class Book {
  String title;
  String author;
  double price;

  Book(this.title, this.author, this.price);

  double discountedPrice(double discountPercent) {
    return price - (price * discountPercent / 100);
  }

  void printInfo() {
    print("Title: $title ,Author: $author ,Price: $price");

  }
}
class Employee {
  String name;
  double salary;
  Employee(this.name, this.salary);

}
class Manager extends Employee {
  String department ;
  Manager (String name,double salary,this.department):super(name,salary);
  void printInfo(){
    print("Name:$name,Salary:$salary,Department:$department");
  }
}
class Developer extends Employee{
  String programmingLanguage ;
  Developer(String name,double salary,this.programmingLanguage):super(name,salary);
  void printInfo(){
    print("Name:$name,Salary:$salary,Programming Language:$programmingLanguage");
  }
}
abstract class Appliance{
  void trunOn();
  void trunOff();
}
class Fan extends Appliance{
  @override
  void trunOn() {
    print("Fan is now running");
  }

  @override
  void trunOff() {
    print("Fan is now stopped");
  }
}
class Light extends Appliance{
  @override
  void trunOn() {
    print("Light is switched on");
  }

  @override
  void trunOff() {
    print("Light is switched off");
  }
}
